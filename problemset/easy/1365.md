## How Many Numbers Are Smaller Than the Current Number / [有多少小于当前数字的数字]()

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var smallerNumbersThanCurrent = function(nums) {
  let arr = [...nums].sort((a, b) => b - a); // big => small
  let map = {};
  let l = arr.length;
  for (let i = 0; i < l; i++) { // 8 3 2 2 1
    if (arr[i] === arr[i+1]) {
      continue;
    } else { // arr[i] > arr[i+1] 
      map[arr[i]] = l - i - 1;
    }
  }
  return nums.map(i => map[i])
};
```
